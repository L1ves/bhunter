# bhunter
## Зависимости:
python3-paramiko <br/>
python3-libtmux <br/>
tmux <br/>
## Установка:
sudo make install <br/>
sudo make keygen <br/>
## Запуск:
sudo bhunter <br/>
  или: <br/>
sudo bhunter-ts <br/>  
sudo tmux attach -t bhunter <br/>
  или: <br/>
systemctl enable bhunter <br/>
systemctl start bhunter <br/>
